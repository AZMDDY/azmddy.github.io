二叉树的递归式遍历和迭代式遍历。

<!-- more -->

二叉树的递归式遍历和迭代式遍历都只需要渐进的时间复杂度，相对而言，前者更加简明，但时间和空间复杂度的常系数相对迭代式更大。

## 先序遍历

### 递归式

```cpp
#include <vector>

using namespace std;

// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

void Dfs(TreeNode* node, vector<int>& res)
{
    if (node == nullptr) {
        return;
    }
    res.push_back(node->val);
    Dfs(node->left, res);
    Dfs(node->right, res);
}

vector<int> PreorderTraversal(TreeNode* root)
{
    vector<int> res;
    Dfs(root, res);
    return res;
}
```

### 迭代式

```cpp
#include <vector>
#include <stack>
using namespace std;

// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

vector<int> PreorderTraversal(TreeNode* root)
{
    vector<int> res;
    stack<TreeNode*> stk;
    while (root != nullptr) {
        while (root != nullptr) {
            res.push_back(root->val);
            if (root->right != nullptr) {
                stk.push(root->right);
            }
            root = root->left;
        }
        if (stk.empty()) {
            break;
        }
        root = stk.top();
        stk.pop();
    }
    return res;
}
```

## 中序遍历

### 递归式

```cpp
#include <vector>

using namespace std;

// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

void Dfs(TreeNode* node, vector<int>& res)
{
    if (node == nullptr) {
        return;
    }
    Dfs(node->left, res);
    res.push_back(node->val);
    Dfs(node->right, res);
}

vector<int> InorderTraversal(TreeNode* root)
{
    vector<int> res;
    Dfs(root, res);
    return res;
}
```

### 迭代式

```cpp
#include <vector>
#include <stack>

using namespace std;

// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

vector<int> InorderTraversal(TreeNode* root)
{
    vector<int> res;
    stack<TreeNode*> stk;
    while (root != nullptr || !stk.empty()) {
        while (root != nullptr) {
            stk.push(root);
            root = root->left;
        }
        root = stk.top();
        stk.pop();
        res.push_back(root->val);
        root = root->right;
    }
    return res;
}
```

## 后序遍历

### 递归式

```cpp
#include <vector>

using namespace std;

// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

void Dfs(TreeNode* node, vector<int>& res)
{
    if (node == nullptr) {
        return;
    }
    Dfs(node->left, res);
    Dfs(node->right, res);
    res.push_back(node->val);
}

vector<int> PostorderTraversal(TreeNode* root)
{
    vector<int> res;
    Dfs(root, res);
    return res;
}
```

### 迭代式

```cpp
#include <vector>
#include <stack>
#include <algorithm>

using namespace std;

// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

vector<int> PostorderTraversal(TreeNode* root)
{
    vector<int> res;
    stack<TreeNode*> stk;
    if (root == nullptr) {
        return res;
    }
    stk.push(root);
    while (!stk.empty()) {
        auto node = stk.top();
        stk.pop();
        if (node != nullptr) {
            res.push_back(node->val);
            stk.push(node->left);
            stk.push(node->right);
        }
    }
    reverse(res.begin(), res.end());
    return res;
}
```

## 层次遍历

```cpp
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode* left;
    TreeNode* right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

vector<int> LevelTraversal(TreeNode* root)
{
    vector<int> res;
    queue<TreeNode*> que;
    if (root == nullptr) {
        return res;
    }
    que.push(root);
    while (!que.empty()) {
        auto node = que.front();
        que.pop();
        res.push_back(node->val);
        if (node->left != nullptr) {
            que.push(node->left);
        }
        if (node->right != nullptr) {
            que.push(node->right);
        }
    }
    return res;
}
```
